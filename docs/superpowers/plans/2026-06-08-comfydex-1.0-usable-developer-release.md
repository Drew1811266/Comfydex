```python
def test_validate_release_package_requires_1_0_install_assets(tmp_path):
    root = Path(__file__).parents[1]
    package_root = tmp_path / "package"
    shutil.copytree(
        root,
        package_root,
        ignore=shutil.ignore_patterns(".git", "__pycache__", ".pytest_cache", "target", "node_modules", "dist", "vite.*.log"),
    )
    (package_root / "scripts" / "install_windows.ps1").unlink()

    errors = validate_release_package(package_root)

    assert any("scripts/install_windows.ps1" in error for error in errors)
```
